To test nominatim you either need to configure your webserver to act as a
reverse proxy for nominatim.openstreetmap.org or you need to create a
controller containing the code below in the pylons project. Please be aware
that this code is not meant to be used in an production environment!

from pylons import request, response, session, tmpl_context as c
from novam.lib.base import BaseController

from urllib import urlretrieve

class NominatimController(BaseController):
	def index(self):
		f = urlretrieve("http://nominatim.openstreetmap.org/?" + request.query_string)[0]
		fh = open(f, "rb")
		response.content_type = "text/xml; charset=utf-8";
		return fh.read()
